async function addProductBatch(sku, qty, customPrice) {
  // Click Add Products
  await evalJs(`
    (function() {
      var btn = document.querySelector('#add_products');
      if (btn) btn.click();
      return 'ok';
    })()
  `);
  await sleep(2000);
  
  // Search SKU
  await evalJs(`
    (function() {
      var skuInput = document.querySelector('#sales_order_create_search_grid_filter_sku');
      if (!skuInput) return 'no input';
      skuInput.value = '${sku}';
      skuInput.dispatchEvent(new Event('input', {bubbles: true}));
      var searchBtn = document.querySelector('#sales_order_create_search_grid button.action-secondary');
      if (searchBtn) searchBtn.click();
      return 'searched';
    })()
  `);
  await sleep(3000);
  
  // Add to cart
  await evalJs(`
    (function() {
      var rows = document.querySelectorAll('#sales_order_create_search_grid_table tbody tr');
      for (var row of rows) {
        var cells = row.querySelectorAll('td');
        for (var cell of cells) {
          if (cell.textContent.trim() === '${sku}') {
            sales_order_create_search_gridJsObject.trOnClick(row);
            return 'added';
          }
        }
      }
      return 'not found';
    })()
  `);
  await sleep(2000);
  
  // Configure qty and price
  await evalJs(`
    (async function() {
      var rows = document.querySelectorAll('#order-items_grid tbody tr');
      var lastRow = rows[rows.length - 1];
      
      var qtyInput = lastRow.querySelector('input[name*="[qty]"]');

        qtyInput.dispatchEvent(new Event('change', {bubbles: true}));

      await new Promise(r => setTimeout(r, 500));
      
      var priceInput = lastRow.querySelector('input[id*="custom_price"]');
      if (priceInput) {
        priceInput.value = '${customPrice}';
        priceInput.dispatchEvent(new Event('change', {bubbles: true}));
      }
      
      var buttons = lastRow.querySelectorAll('button');
      for (var btn of buttons) {
        if (btn.textContent.trim() === 'Apply') {
          btn.click();
          break;
        }
      }
      return 'configured';
    })()
  `);
  await sleep(1500);
}

(function() {
  var items = document.querySelectorAll('#order-items_grid tbody tr');
  var total = 0;
  
  for (var item of items) {
    var cells = item.querySelectorAll('td');
    // Find subtotal column (usually last numeric column)
    for (var cell of cells) {
      var text = cell.textContent.trim();
      if (text.startsWith('₹')) {
        var num = parseFloat(text.replace('₹', '').replace(/,/g, ''));
        if (!isNaN(num) && num > 0) total += num;
      }
    }
  }
  
  return JSON.stringify({
    itemCount: items.length,
    estimatedTotal: total
  });
})()
